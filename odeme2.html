function addMonthsKey(key, delta){
  const [y,m] = String(key).split("-").map(Number);

}

function bakiyeKisa(net){
  net = Number(net || 0);
  if(net > 0) return `${net} TL Fazla`;
  if(net < 0) return `${Math.abs(net)} TL Borç`;
  return "0 TL";
}

function bakiyeYazi(net){
  if(net > 0){
    return {
      text: `${net} TL Fazla`,
      cls: "balanceGood",
      desc: "Sonraki dönem aidatlarından otomatik düşülür."
    };
  }

  if(net < 0){
    return {
      text: `${Math.abs(net)} TL Borç`,
      cls: "balanceBad",
      desc: "Ödenmeyen tutar sonraki aya devreder."
    };
  }

  return {
    text: "0 TL",
    cls: "balanceNeutral",
    desc: "Borç veya fazla ödeme yok."
  };
}

  document.getElementById(aciklamaId).textContent = info.desc;
}

    donemler.push(cursor);
    cursor = addMonthsKey(cursor, 1);
  }

  let bakiye = 0; // + fazla, - borç
  let oncekiDevirAktif = 0;
  let aktifIslemNet = -aktifDonemTutar;

    if(key === aktifDonem){
      oncekiDevirAktif = oncekiBakiye;
      aktifIslemNet = odenenTutar - donemTutar;
      aktifEtiket = bakiye >= 0
        ? {text:"Borç Yok", cls:"odendi"}
        : {text:"Borç Var", cls:"gecikti"};
    }

    const odenenSatir = odenenTutar > 0 ? `Bu Ay Ödenen: ${odenenTutar} TL<br>` : `Bu Ay Ödenen: 0 TL<br>`;

        <div class="historyLeft">

          <div class="historyMoney">
            ${devirSatir}

      <div class="warningBox">
