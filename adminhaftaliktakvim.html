function randevuSnapIsle(snap, koleksiyonAdi, arsivKaydiMi) { if (!snap) return; snap.forEach(function(docItem) { const gorulenKey = koleksiyonAdi + "/" + docItem.id; if (gorulenDoclar.has(gorulenKey)) return; gorulenDoclar.add(gorulenKey); const data = docItem.data() || {}; // 🔥 EKLENEN KISIM (EN KRİTİK) const durum = String(data.status || data.durum || "").toLowerCase(); if (durum === "iptal" || durum === "cancelled" || durum === "canceled") { return; // 👉 artık DOLU sayılmaz } const tarih = kayitTarih(data); const saat = kayitSaat(data); const isim = kayitAd(data) || "-"; const telefon = kayitTelefon(data) || "-"; const hizmet = kayitHizmet(data) || "-"; if (!tarih || !saat) return; if (tarih < ilkGun || tarih > sonGun) return; const key = tarih + "_" + saat; randevuMap[key] = { isim, telefon, hizmet, gecmis: arsivKaydiMi ? true : randevuGecmisMi(data, simdi) }; }); }
